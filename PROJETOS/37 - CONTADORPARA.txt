algoritmo "contador1a10"
//Programa que utiliza estrutura de repetição "Para", para contar
//sequencias de números com diferentes "passos".

var
   c: inteiro

inicio
   Para c <- 1 ate 10 passo 1 faca
      Escreval(c)
   Fimpara
   
   Escreval(" ||")
   Para c <- 1 ate 10 passo 2 faca
      Escreval(c)
   Fimpara
   
   Escreval(" ||")
   Para c <- 10 ate 1 passo -1 faca
      Escreval(c)
   Fimpara

fimalgoritmo
