//Programa que, utiliza função, para verifica se um numero e PAR ou IMPAR.

var
  n: Inteiro
  r: Caractere

Funcao ParImpar(v: Inteiro): Caractere
inicio
   Se (v % 2 = 0) entao
      Retorne "PAR"
   senao
      Retorne "IMPAR"
   FimSe
   
FimFuncao
